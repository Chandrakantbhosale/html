
// 86 -  Implemen the nor operator


// age in between less than and greater than
// all other are access

db.users.find({
     $nor:[
    {age : {$lt: 30}},
    {age : {$gt: 50}}
    ]
})

// city not search

db.users.find({
   $nor: [
   {city: "mumbai" },
   {city: "banglore"},
   {city: "Wisokyburgh"},
   {city: "Bell City"}
   ]
})
